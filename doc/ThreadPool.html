def initialize(size)
  @queue = Queue.new
  @threads = []

  size.times do
    @threads << Thread.new do
      catch(:exit) do
        loop do
          job, args = @queue.pop
          job.call(*args)
        end
      end
    end
  end
end

def shutdown
  @threads.size.times do
    schedule { throw :exit }
  end

  @threads.each(&:join)
end
